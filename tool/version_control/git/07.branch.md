# Branch

# create branch

> $ git branch `<branch_name>`

# branch list

> $ git branch

> $ git branch `-v`

> $ git branch `-r` `-v`

# checkout branch

> $ git checkout `<branch_name>`

# checkout create brach

> $ git checkout -b `<branch_name>`

# branch merge

> $ git merge `<branch_name>`

# branch delete

> $ git branch `-d <branch_name>`

# rebase

> $ git rebase master `--continue`
